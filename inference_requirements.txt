# Local Prithvi sen1floods11 inference dependencies.
#
# Python 3.12-oriented local Prithvi inference setup.
#
# The original IBM/NASA Prithvi sen1floods11 release used the legacy
# MMSegmentation 0.x / MMCV 1.x stack and hls-foundation-os pins timm==0.4.12.
# That conflicts with modern torchgeo/terratorch stacks, which need timm>=1.x.
#
# For Python 3.12 in your existing environment, install a fork of
# hls-foundation-os where setup.py does NOT pin timm or pull mmsegmentation.
#
# CPU example:
#   conda create -n prithvi-flood python=3.12 -y
#   conda activate prithvi-flood
#   pip install torch torchvision --index-url https://download.pytorch.org/whl/cpu
#   pip install "setuptools<70" wheel
#   pip install --no-build-isolation -r inference_requirements.txt
#
# CUDA example: choose the index URL that matches your driver/CUDA runtime.
#   conda create -n prithvi-flood python=3.12 -y
#   conda activate prithvi-flood
#   pip install torch torchvision --index-url https://download.pytorch.org/whl/cu118
#   pip install "setuptools<70" wheel
#   pip install --no-build-isolation -r inference_requirements.txt

setuptools<70
numpy
rasterio
matplotlib
huggingface_hub
timm>=1.0.15
opencv-python-headless
rioxarray

# Legacy Prithvi/MMseg stack.
mmcv==1.7.2
mmsegmentation==0.30.0
